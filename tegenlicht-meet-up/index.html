def selection_sort(list)
  return list if list.size <= 1

  0.upto(list.length-2) do |i|
    min = i
    (i+1).upto(list.length-1) do |j| 
      min = j if list[j] < list[min] 
    end
    if i != min
      list[i], list[min] = list[min], list[i] 
    end
  end

  list
end
